            Console.WriteLine("Enter the chemistry mark : ");

            total2 = math + phy;
            Console.WriteLine("Total mark of physic , math and chemistry : " +total1);
            Console.WriteLine("Total mark of math physic : " +total2);

            if (math >= 65)
                if (phy >= 55)
                    if (chem >= 50)
                        if ((math + phy + chem) >= 190 || (math + phy) >= 140)
