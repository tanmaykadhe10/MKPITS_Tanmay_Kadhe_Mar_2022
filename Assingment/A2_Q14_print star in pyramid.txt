using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace A2_Q14_print_star_in_pyramid
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int i, j, n, rows, k, m = 1;


            Console.Write("input number of rows : ");
            rows = Convert.ToInt32(Console.ReadLine());
            n = rows + 4 - 1;
            for (i = 1; i <= rows; i++)
            {
                for (k = n; k >= 1; k--)
                {
                    Console.Write(" ");
                }
                for (j = 1; j <= i; j++)
                    Console.Write("* ", m++);
                Console.Write("\n");
                n--;
            }
            Console.ReadLine();
        }
    }
}
