    internal class Program

        {
            int n,i,d=0;
            Console.WriteLine("Enter the number");
            n = Convert.ToInt32(Console.ReadLine());


			i = 1;

			while (i <= n)
			{
				if (n % i == 0)
				{
					d == d + 1;
					break;
				}
				i = i + 1;
			}
			if (d == 2)
			{
				Console.WriteLine("Enter Number is Prime !");
			}
			else
			{
			    Console.WriteLine("Enter Number is Not Prime Number !");
			}

			Console.ReadLine();
        }
    }
